- `./etc/default/unlim8ted`

- `./etc/xdg/autostart/unlim8ted-chromium.desktop`
- `./opt/unlim8ted/bin/launch-unlim8ted.sh`
- `./opt/unlim8ted/bin/start-chromium.sh`
